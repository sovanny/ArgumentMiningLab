# creating directed graphs and adding nodes and edges
for pair in root.iter('pair'):
    #print('Adding argument pair id: ' + pair.attrib['id'])
    node_t_id = int(pair[0].attrib['id'])
    node_h_id = int(pair[1].attrib['id'])
    if(pair.attrib['topic'] in list_of_topics):
        index = list_of_topics.index(pair.attrib['topic'])
        list_of_graphs[index].add_node(node_t_id, text = pair[0].text)
        list_of_graphs[index].add_node(node_h_id, text = pair[1].text)
        list_of_graphs[index].add_edge(node_t_id, node_h_id, entailment = int(pair.attrib['entailment']))
    else:            
        list_of_topics.append(pair.attrib['topic'])
        list_of_graphs.append(nx.DiGraph(topic=pair.attrib['topic']))
        list_of_graphs[-1].add_node(node_t_id, text = pair[0].text)
        list_of_graphs[-1].add_node(node_h_id, text = pair[1].text)
        list_of_graphs[-1].add_edge(node_t_id, node_h_id, entailment = int(pair.attrib['entailment']))

In [150]:
essay_graphs = []
claimstances =[]
for ind in range(len(essays)):
    essay_graphs.append(nx.DiGraph(topic = topics[ind]))
    majorclaim_id = 0;
    for line in essays[ind]:
        if(line.startswith('A')):
            claimstances.append(re.split(r' ', re.split(r'\t+', line)[1])[1:])                      
        if(line.startswith('T')):
            split_line = re.split(r'\t+', line)
            node_id = int(split_line[0][1:])
            [ent, start_ind, end_ind] = re.split(r' ', split_line[1])
            if(ent == 'MajorClaim'): majorclaim_id = node_id
            txt = split_line[2]
            essay_graphs[ind].add_node(node_id, entity = ent, start = start_ind, end = end_ind, text = txt)
        if(line.startswith('R')):
            [rel, arg1, arg2] = re.split(r' ', re.split(r'\t+', line)[1])
            if(rel == 'supports'): rel = 1
            else: rel = 0
            node_1 = int(arg1[6:])
            node_2 = int(arg2[6:])
            essay_graphs[ind].add_edge(node_1, node_2, entailment = rel)
    for stance in claimstances:
        if(stance[1] == 'For'):
            essay_graphs[ind].add_edge(int(stance[0][1:]), majorclaim_id, entailment = 1)
        else: essay_graphs[ind].add_edge(int(stance[0][1:]), majorclaim_id, entailment = 0)
